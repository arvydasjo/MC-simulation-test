########### Configuration of GCP with VSCODE terminal ############

# GCP config
gcloud auth login
gcloud config set project <PROJECT_ID>
gcloud config set compute/zone europe-west1-b

# (OPTIONAL) sukurti VM jei dar nėra
gcloud compute instances create mc-vm \
    --machine-type=n2-standard-4 \
    --image-family=debian-11 \
    --image-project=debian-cloud

# pakeisti VM tipą (VM turi būti STOPPED!)
gcloud compute instances stop mc-vm --zone=europe-west1-b
gcloud compute instances set-machine-type mc-vm \
    --zone=europe-west1-b \
    --machine-type=n2-standard-4

# start/stop VM
gcloud compute instances start mc-vm --zone=europe-west1-b
gcloud compute instances stop mc-vm --zone=europe-west1-b

# check status
gcloud compute instances list

# connect via SSH
gcloud compute ssh mc-vm --zone=europe-west1-b
